<script>
let votes = JSON.parse(localStorage.getItem('voteData')) || {p1:0,p2:0,p3:0};
let voted = localStorage.getItem('hasVoted') === 'true';

function updateResult(){
document.getElementById('result').innerHTML = `
<div class="result-item"><span>张三</span><span>${votes.p1} 票</span></div>
<div class="result-item"><span>李四</span><span>${votes.p2} 票</span></div>
<div class="result-item"><span>王五</span><span>${votes.p3} 票</span></div>
`}

function disableAll(){document.querySelectorAll('.vote-btn').forEach(b=>b.disabled=true)}
if(voted) disableAll(); updateResult();

document.querySelectorAll('.vote-btn').forEach(btn=>{
btn.onclick = ()=>{
if(voted) return;
const id = btn.dataset.id; votes[id]++;
localStorage.setItem('voteData',JSON.stringify(votes));
localStorage.setItem('hasVoted','true');
voted = true; disableAll(); updateResult();
alert('投票成功！感谢参与～')
}})
</script>
